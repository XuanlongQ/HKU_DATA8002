import networkx as nx
import numpy as np
import matplotlib.pyplot as plt

# Parameters
N = 200  # number of nodes
p = 0.01  # probability of edge
num_networks = 1000

# Initialize lists to store results
num_edges = []
avg_clustering = []
avg_path_length = []
num_components = []

print("Generating 1000 Erds-Rnyi networks...")

# Generate 1000 random networks and compute measures
for i in range(num_networks):
    if (i + 1) % 100 == 0:
        print(f"Processed {i + 1}/{num_networks} networks")
    
    # Generate Erds-Rnyi random graph
    G = nx.erdos_renyi_graph(N, p)
    
    # (a) Number of edges
    num_edges.append(G.number_of_edges())
    
    # (b) Average clustering coefficient
    avg_clustering.append(nx.average_clustering(G))
    
    # (c) Average shortest path length
    # Need to handle disconnected graphs - compute for largest component
    if nx.is_connected(G):
        avg_path_length.append(nx.average_shortest_path_length(G))
    else:
        # For disconnected graphs, compute on largest connected component
        largest_cc = max(nx.connected_components(G), key=len)
        subgraph = G.subgraph(largest_cc)
        if len(largest_cc) > 1:  # Avoid single-node components
            avg_path_length.append(nx.average_shortest_path_length(subgraph))
        else:
            avg_path_length.append(0)  # Single node has path length 0
    
    # (d) Number of connected components
    num_components.append(nx.number_connected_components(G))

# Convert to numpy arrays for easier handling
num_edges = np.array(num_edges)
avg_clustering = np.array(avg_clustering)
avg_path_length = np.array(avg_path_length)
num_components = np.array(num_components)

# Create plots
fig, axes = plt.subplots(2, 2, figsize=(15, 12))

# (a) Number of edges

import networkx as nx
import numpy as np
import matplotlib.pyplot as plt
from collections import defaultdict

def percolation_analysis(G, attack_type="targeted"):
    nodes = list(G.nodes())
    
    if attack_type == "targeted":
        degrees = dict(G.degree())
        sorted_nodes = sorted(nodes, key=lambda x: degrees[x], reverse=True)
    else:  # random
        np.random.shuffle(nodes)
        sorted_nodes = nodes
    
    components = []
    largest_component_sizes = []
    
    G_copy = G.copy()
    
    for i, node in enumerate(sorted_nodes):
        if node in G_copy:
            G_copy.remove_node(node)
        
        connected_components = list(nx.connected_components(G_copy))
        num_components = len(connected_components)
        
        if connected_components:
            largest_size = max(len(comp) for comp in connected_components)
        else:
            largest_size = 0
            
        components.append(num_components)
        largest_component_sizes.append(largest_size)
    
    return components, largest_component_sizes

G_ER = nx.erdos_renyi_graph(1000, 0.01)
G_BA = nx.barabasi_albert_graph(1000, 5)

# Targeted attack
er_comp_targeted, er_lcc_targeted = percolation_analysis(G_ER, "targeted")
ba_comp_targeted, ba_lcc_targeted = percolation_analysis(G_BA, "targeted")

# Random attack
er_comp_random, er_lcc_random = percolation_analysis(G_ER, "random")
ba_comp_random, ba_lcc_random = percolation_analysis(G_BA, "random")
